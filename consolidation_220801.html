# 3 Convert all columns to numerical
# 3.1 Clean the columns with .str.replace()
# df_SPX['Date'].str.replace('','')

df_SPX['High'] = df_SPX['High'].str.replace(',','')
df_SPX['Low'] = df_SPX['Low'].str.replace(',','')
